#!/usr/bin/env node
(function() {
  var argv, argv_, config, dir, extrafs, fs, inspect, log, optimist, output, path, ready;

  dir = console.dir, log = console.log;

  inspect = require('util').inspect;

  path = require('path');

  fs = require('fs');

  extrafs = require('../node_modules/fs-extra');

  ready = require('../lib/');

  output = require('../lib/output');

  optimist = require('optimist').usage('\nUsage: $0 [FILES OR DIRECTORIES] [options]').options('o', {
    alias: 'output'
  }).describe('o', 'The file or directory in which to write the output').option('c', {
    alias: 'config'
  }).describe('c', 'Specify a config.json file').options('i', {
    alias: 'ignore'
  }).describe('i', 'Ignore these files from JSHint but output them in the aggregated file').options('k', {
    alias: 'keep'
  }).describe('k', 'Keep individual minified files').describe('no-recursive', 'Don\'t recurse in sub-directories').options('h', {
    alias: 'help'
  }).describe('h', 'Display this help').options('v', {
    alias: 'version'
  }).describe('v', 'Display the current version');

  argv = optimist.argv;

  if (argv.v != null) {
    require('../lib/version')(function(version) {
      return log(version);
    });
    return;
  }

  if (argv._.length === 0 || (argv.help != null)) {
    return optimist.showHelp();
  }

  if (argv.ignore != null) {
    argv.ignore = argv.ignore.split(' ');
  }

  argv.i = argv.ignore;

  if (argv.k && argv.o) {
    ready.on('file.uglify', function(file, uglify) {
      var minFilename;
      minFilename = path.basename(file).replace(/\.js$/, '.min.js');
      return output.writeToDir(uglify.code, minFilename, argv, function(err) {
        if (err != null) {
          console.error(err);
          return process.exit(1);
        }
      });
    });
  }

  if (argv.c != null) {
    argv_ = argv._;
    config = fs.readFileSync(argv.c);
    argv = JSON.parse(config);
    argv._ = argv_;
  }

  if (argv._.length > 0) {
    ready.compile(argv._, argv, function(err, minified) {
      if (err != null) {
        console.error(err);
        process.exit(1);
      }
      return output.write(minified, argv.output, function(err) {
        if (err != null) {
          console.error(err);
          return process.exit(1);
        }
      });
    });
  }

}).call(this);
