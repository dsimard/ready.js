(function() {
  var inspect, reporter, _;

  inspect = require('util').inspect;

  _ = require('../node_modules/underscore');

  reporter = function(filename, errors) {
    var message;
    message = ["" + errors.length + " error" + (errors.length > 1 ? 's' : '') + " in '" + filename + "'\n"];
    errors = _.compact(errors).map(function(e) {
      var line;
      line = (e.line != null) && (e.character != null) ? "" + e.line + ":" + e.character + " : " : "";
      return "" + line + e.reason;
    });
    return message.concat(errors).join('\n');
  };

  module.exports = {
    reporter: reporter
  };

}).call(this);
