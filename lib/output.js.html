(function() {
  var fileExists, fs, log, minimatch, path, r;

  path = require('path');

  fs = require('fs');

  fileExists = fs.exists || path.exists;

  minimatch = require('../node_modules/minimatch');

  log = console.log;

  r = {
    write: function(content, destination, callback) {
      if (destination == null) {
        return console.log(content);
      }
      if (minimatch(destination, '**/*.js')) {
        return r.writeToFile(content, destination, callback);
      } else {
        return r.writeToFile(content, "" + destination + "/all.js", callback);
      }
    },
    writeToFile: function(content, destination, callback) {
      return fs.writeFile(destination, content, function(err) {
        if (err != null) {
          return callback(err);
        }
        return callback();
      });
    },
    writeToDir: function(content, filename, options, callback) {
      var output, outputDir;
      output = options.output;
      outputDir = output;
      if (minimatch(output, '**/*.js')) {
        outputDir = path.dirname(output);
      }
      return r.writeToFile(content, "" + outputDir + "/" + filename, callback);
    }
  };

  module.exports = r;

}).call(this);
